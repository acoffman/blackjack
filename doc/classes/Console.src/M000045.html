        def display_result(result,bet)
                case result
                when :lose
                        puts "Hand loses, -$#{bet}"
                when :win
                        puts "Hand wins, +$#{bet}"
                when :blackjack
                        puts "Blackjack! Hand wins, +$#{Integer(1.5*bet)}"
                when :push
                        puts "Push, no change."
                when :bust
                        puts "Bust! Hand loses, -$#{bet}"
                end
        end
