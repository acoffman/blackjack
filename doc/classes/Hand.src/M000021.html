        def to_s
                string_val = ""
                @cards.each do |current_card|
                        string_val << current_card.to_s
                        string_val << ", "
                end
                if hand_value.length == 0
                        string_val << "BUST"
                else
                        string_val << "(" + hand_value.join(",") + ")"
                end
                return string_val
        end
